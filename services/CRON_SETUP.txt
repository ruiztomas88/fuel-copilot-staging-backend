#

0 */6 * * * /Users/tomasruiz/Desktop/Fuel-Analytics-Backend/services/backup_mysql.sh >> /Users/tomasruiz/Desktop/Fuel-Analytics-Backend/logs/cron_backup.log 2>&1
#
# ============================================================================
# NOTAS:
# - Los backups se guardan en: /Users/tomasruiz/Desktop/Fuel-Analytics-Backend/backups/

# - El log de cron se guarda en: logs/cron_backup.log

# crontab -l | grep backup
